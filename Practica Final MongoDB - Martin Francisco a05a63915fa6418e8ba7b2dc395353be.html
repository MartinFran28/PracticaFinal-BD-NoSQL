1. Seleccionar el apellido, oficio, salario, numero de departamento y su nombre de todos los empleados cuyo salario sea mayor de 300000

db.EMP.aggregate([
  {
    $lookup: {
      from: 'DEPT',
      localField: 'DEPT_NO',
      foreignField: 'DEPT_NO',
      pipeline: [
        { $project: { _id: 0, DEPT_NOMBRE: 1, DEPT_NO: 1 } }
      ],
      as: 'DEPARTAMENTO'
    }
  },
  {
    $addFields: {
      SALARIO_ANUAL: { $multiply: ['$SALARIO', 12] }
    }
  },
  {
    $match: {
      SALARIO_ANUAL: { $gt: 3000000 }
    }
  },
  {
    $project: {
      _id: 0,
      APELLIDO: 1,
      OFICIO: 1,
      SALARIO: 1,
      DEPT_NO: { $arrayElemAt: ['$DEPARTAMENTO.DEPT_NO', 0] },
      DEPT_NOMBRE: { $arrayElemAt: ['$DEPARTAMENTO.DEPT_NOMBRE', 0] }
    }
  }
]);

12. Visualizar el nombre de los empleados de la plantilla junto con el nombre de la sala, el nombre del hospital y el número de camas libres de cada una de ellas

db.PLANTILLA.aggregate([
  {
    $lookup: {
      from: 'HOSPITAL',
      localField: 'HOSPITAL_COD',
      foreignField: 'HOSPITAL_COD',
      pipeline: [
        {
          $lookup: {
            from: 'SALA',
            localField: 'HOSPITAL_COD',
            foreignField: 'HOSPITAL_COD',
            as: 'SALA'
          }
        }
      ],
      as: 'HOSPITAL'
    }
  },
  {
    $lookup: {
      from: 'SALA',
      localField: 'HOSPITAL_COD',
      foreignField: 'HOSPITAL_COD',
      as: 'SALA'
    }
  },
  {
    $unwind: '$HOSPITAL'
  },
  {
    $addFields: {
      SALA: {
        $map: {
          input: '$HOSPITAL.SALA',
          as: 'hospitalSala',
          in: {
            $cond: {
              if: { $eq: ['$$hospitalSala.SALA_COD', '$SALA_COD'] },
              then: '$$hospitalSala',
              else: null
            }
          }
        }
      }
    }
  },
  {
    $addFields: {
      HOSPITAL_SALA: {
        $filter: {
          input: '$SALA',
          as: 'hospitalSala',
          cond: { $ne: ['$$hospitalSala', null] }
        }
      }
    }
  },
  {
    $project: {
      _id: 0,
      APELLIDO: 1,
      HOSPITAL: '$HOSPITAL.NOMBRE',
      SALA: { $arrayElemAt: ['$HOSPITAL_SALA.NOMBRE', 0] },
      NO_CAMAS: { $arrayElemAt: ['$HOSPITAL_SALA.NUM_CAMA', 0] }
    }
  }
]);

20. Queremos saber el apellido, oficio, salario y número de departamento de los empleados con salario mayor que el mejor salario del departamento 30

db.EMP.aggregate([
  { $match: { DEPT_NO: 30 } },
  { $group: { _id: null, SALARIO_MAXIMO: { $max: "$SALARIO" } } },
  { $lookup: {
      from: "EMP",
      let: { salarioMaximo: "$SALARIO_MAXIMO" },
      pipeline: [
        { $match: { $expr: { $gt: ["$SALARIO", "$$salarioMaximo"] } } },
        { $project: { APELLIDO: 1, OFICIO: 1, SALARIO: 1, DEPT_NO: 1 } }
      ],
      as: "empleados"
    }
  },
  { $unwind: "$empleados" },
  { $replaceRoot: { newRoot: "$empleados" } },
  { $project: { _id: 0, APELLIDO: 1, OFICIO: 1, SALARIO: 1, DEPT_NO: 1 } }
]);
